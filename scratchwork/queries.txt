SELECT ISSUES.SUMMARY, ISSUES.DESCRIPTION, ISSUESLIST.NAME
FROM ISSUES
JOIN ISSUESLIST
ON ISSUES.ISSUESLIST_ID = ISSUESLIST.ID;


SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U
ON I.USERS_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, U.ID, COUNT(*) AS ISSUE_COUNT
FROM ISSUES I
JOIN USERS U
ON I.USERS_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID
HAVING COUNT(*) > 0;
